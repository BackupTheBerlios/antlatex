\section{Entwicklungsgeschichte}
\label{sec:entwicklungsgeschichte}

\begin{description}

\item[0.0.9\_1 (2007-09-25)]\ \\[-2em]
   \begin{itemize}
   \item Integration in antLib ergänzt und entsprechend beschrieben (Hinweis von Simone Tripodi).
   \end{itemize}
\item[0.0.9 (2006-11-20)]\ \\[-2em]
   \begin{itemize}
   \item Dokumentation wegen \code{auxDir}, \code{outputDir} und \verb|\include{EinVerzeichnis/Datei}| überarbeitet.
   \item Offizielle Version
   \end{itemize}
\item[0.0.8\_2 (2006-09-09)]\ \\[-2em]
   \begin{itemize}
   \item Neu: \code{fileExtensions} ergänzt, damit können jetzt
     beliebige Dateiendungen für \LaTeX-Dateien benutzt werden. Vom
     normalen Schema sollte dennoch nicht abgewichen werden.
   \item Fix: \code{command} wurde nicht mehr durchgereicht.
   \item Fix: Bei den Zeichenkettenparametern wurde die Variablenauswertung nicht berücksichtigt
   \item Fix: Parameter \code{outputDir} und \code{auxDir} waren falsch beschrieben.
   \end{itemize}
\item[0.0.8\_1 (2006-04-09)]\ \\[-2em]
   \begin{itemize}
   \item Neu: Verarbeitet jetzt auch \code{*.dtx} und \code{*.ins}-Dateien
     und lässt sich somit in die Installationsroutine für
     \LaTeX-Distributionen einbauen!
   \end{itemize}
\item[0.0.8 (2006-04-01)]\ \\[-2em]
   \begin{itemize}
   \item Internes Release
   \end{itemize}
\item[0.0.7\_2 (2006-03-10)]\ \\[-2em]
   \begin{itemize}
   \item Fehlerbereinigung. Durch die Änderung der \BibTeX{}- bzw.
     MakeIndex-Aufrufe waren die entsprechenden Attribute nicht
     unbedingt gesetzt.
   \end{itemize}
\item[0.0.7 (2006-02-09)]\ \\[-2em]
  \begin{itemize}
  \item Es können jetzt mehrere \BibTeX{}-Aufrufe im \LaTeX-Task
    eingetragen werden, mit dem Schalter \code{inloop} kann gesteuert
    werden, ob \BibTeX{} in der \LaTeX-Schleife aufgerufen werden
    soll.
  \end{itemize}
\item[0.0.6\_2 (2006-01-12)]\ \\[-2em]
  \begin{itemize}
  \item \code{path}-Attribut für den Pfad zu (PDF)\LaTeX{} darf nicht
    mit getCanonicalPath() behandelt werden, weil unter Unix einige
    Programme ihre Funktionalität über den Programmnamen erhalten und
    dementsprechend verlinkt sind. (Gefunden von Alphonse Bendt)
  \end{itemize}

\item[0.0.6 (2006-01-09)]\ \\[-2em]
  \begin{itemize}
  \item \code{path}-Attribut für den Pfad zu (PDF)\LaTeX{}
    ergänzt und damit die Semantik von \code{command}
    geändert. (Gefunden von Alphonse Bendt)
  \item Die Argumente \code{output-directory, aux-directory, job-name}
    für \LaTeX{} umgesetzt und die Subaufrufe von \BibTeX, makeindex
    und \GlossTeX{} angepasst. Die restlichen Argumente können jetzt
    mit \code{passThruLaTeXParameters} getrennt mit Semikolons an
    \LaTeX{} weitergegeben werden.
  \item Mit dem Attribut \code{figbib} wird das Package \code{figbib}
    im \LaTeX-Lauf berücksichtigt.
  \item Keine \LaTeX-Datei gefunden bzw. angegeben führt zu einer
    Fehlermeldung wenn \code{verbose} auf \code{on} geschaltet ist.
  \end{itemize}
        
\item[0.0.5\_2 (2006-01-06)]\ \\[-2em]
  \begin{itemize}
  \item replace-Aufruf auf Java 1.4 geändert. (Gefunden von Alphonse Bendt)
  \item Keine \LaTeX-Datei gefunden bzw. angegeben führt zu einer vernünftigen Fehlermeldung.
  \end{itemize}
        
\item[0.0.5 (2005-12-29)]\ \\[-2em]
  \begin{itemize}
  \item Bei den Tasks \code{LaTeX} und \code{LaTeXTask} konnte nur
    eine Datei angegeben werden, jetzt kann auch das Element
    $<$fileset$>$ benutzt werden. (Gefunden von Alphonse Bendt)
  \end{itemize}

\item[0.0.4 (2005-11-26)]\ \\[-2em]
  \begin{itemize}
  \item Es gab zwar die Möglichkeit mit $<$fileset$>$ einen Filter für
    $<$makeindex$>$ zu setzen, doch dieses Element wurde innerhalb
    eines $<$latex$>$ Tasks nicht ausgewertet. (Gefunden von Thomas
    Reuter)
  \item Das XML-basierte \textbf{jxGlossar} wurde mit eingebunden.
  \item Das Element $<$delete$>$ wurde ersetzt durch
    $<$deletetempfile$>$
  \end{itemize}

\item[0.0.3 (2005-09-23)]\ \\[-2em]
  \begin{itemize}
  \item Das Attribut \code{if} stimmte nicht mit der normalen Semantik
    von Ant überein, deswegen wurde jetzt das Attribut \code{run} zum
    Ein/Ausschalten des Tasks eingeführt.
  \end{itemize}

\item[0.0.2 (2005-09-07)]\ \\[-2em]
  \begin{itemize}
  \item Beim Task \code{LaTeX} die Elemente \code{makeindex},
    \code{bibtex} und \code{glosstex} zugelassen um die
    \glqq{}komplexeren\grqq{} Attribute durchreichen zukönnen.
  \item Mit dem \code{if}-Attribut können jetzt die Tasks ein- bzw.
    ausgeschaltet werden.
  \end{itemize}

\item[0.0.1 (2005-09-02)]\ \\[-2em]
  \begin{itemize}
  \item Erste Veröffentlichung
  \end{itemize}

\end{description}
